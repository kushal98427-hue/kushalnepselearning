    if(fill) fill.style.width=(count/80*100)+'%';
    if(text) text.textContent=count+'/80';
    if(subtext){
        if(count===0) subtext.textContent='chapters पढ्न सुरु गर्नुहोस्!';
        else if(count<8) subtext.textContent='राम्रो सुरुवात! पढ्दै जानुहोस् 📚';
        else if(count<29) subtext.textContent='बढिया! तपाईं राम्रोसँग अगाडि बढिरहनुभएको छ 🔥';
        else subtext.textContent='अभूतपूर्व! तपाईं एक dedicated reader हुनुहुन्छ! 🏆';
    }
}

async function submitPaymentProof() {
    
    const fileInput = document.getElementById('payment-screenshot');
    const statusMsg = document.getElementById('payment-status-msg');

    if(!fileInput.files[0]){ alert('Screenshot upload गर्नुहोस्।'); return; }

    const {data:{session}} = await supabaseClient.auth.getSession();
    if(!session){ openAuthModal(); return; }

    const file = fileInput.files[0];
    const fileName = session.user.id + '_' + Date.now() + '_' + file.name;

    const {error: uploadError} = await supabaseClient.storage
        .from('payment-proofs').upload(fileName, file);

    if(uploadError){
        statusMsg.style.cssText='display:block;padding:15px;border-radius:8px;background:#fef2f2;color:#ef4444;';
        statusMsg.textContent = 'Upload error: ' + uploadError.message; return;
    }

    const {error: dbError} = await supabaseClient.from('payment_requests').insert({
        user_id: session.user.id,
        transaction_id: '',
        screenshot_path: fileName,
        status: 'pending'
    });

    if(dbError){
        statusMsg.style.cssText='display:block;padding:15px;border-radius:8px;background:#fef2f2;color:#ef4444;';
        statusMsg.textContent = 'DB error: ' + dbError.message; return;
    }

    // Screenshot को signed URL निकाल्ने (7 दिनसम्म valid)
    const {data: urlData} = await supabaseClient.storage
        .from('payment-proofs').createSignedUrl(fileName, 604800);
    const screenshotUrl = urlData?.signedUrl || 'Supabase Storage मा हेर्नुहोस्';

    // Admin email notification पठाउने
    emailjs.send("service_bzntxad", "template_1bl4hnx", {
        user_email: session.user.email,
        submission_time: new Date().toLocaleString('ne-NP'),
        screenshot_url: screenshotUrl
    });

    statusMsg.style.cssText='display:block;padding:15px;border-radius:8px;background:#f0fdf4;color:#16a34a;font-weight:600;';
    statusMsg.innerHTML = '✅ Payment proof पठाइयो! हामी <strong>24 घण्टाभित्र</strong> verify गरेर access दिन्छौं।';
    document.getElementById('payment-form-section').style.display = 'none';
}

function selectStar(n) {
    selectedRating = n;
    document.querySelectorAll('#star-selector span').forEach((s, i) => {
        s.style.opacity = i < n ? '1' : '0.3';

async function submitReview() {
    const msg = document.getElementById('review-msg');
    if(selectedRating === 0) {
        msg.style.cssText = 'display:block;background:#fef2f2;color:#ef4444;';
        msg.textContent = 'कृपया star rating दिनुहोस्।';
        return;
    }
    const {data:{session}} = await supabaseClient.auth.getSession();
    if(!session) { openAuthModal(); return; }
    const reviewText = document.getElementById('review-text').value.trim();
    const name = session.user.user_metadata?.full_name || session.user.email.split('@')[0];
    const {error} = await supabaseClient.from('book_reviews').upsert({
        user_id: session.user.id,
        rating: selectedRating,
        review_text: reviewText,
        reviewer_name: name
    }, {onConflict: 'user_id'});
    if(error) {
        msg.style.cssText = 'display:block;background:#fef2f2;color:#ef4444;';
        msg.textContent = 'Error: ' + error.message;
        return;
    }
    msg.style.cssText = 'display:block;background:#f0fdf4;color:#16a34a;font-weight:600;';
    msg.textContent = '✅ Review पठाइयो! धन्यवाद।';
    loadReviews();
}

function escapeHtml(str){
    if(!str) return '';
    return String(str)
        .replace(/&/g,'&amp;')
        .replace(/</g,'&lt;')
        .replace(/>/g,'&gt;')
        .replace(/"/g,'&quot;')
        .replace(/'/g,'&#039;');
}

async function loadReviews(){
    const {data} = await supabaseClient.from('book_reviews')
        .select('*').order('created_at',{ascending:false});
    if(!data||data.length===0) return;
    const avg = (data.reduce((s,r)=>s+r.rating,0)/data.length).toFixed(1);
    const avgEl = document.getElementById('avg-score');
    const totalEl = document.getElementById('total-reviews');
    const starsEl = document.getElementById('avg-stars');
    if(avgEl) avgEl.textContent = avg;
    if(totalEl) totalEl.textContent = data.length+' reviews';
    const stars = Math.round(parseFloat(avg));
    if(starsEl) starsEl.textContent = '⭐'.repeat(stars)+'☆'.repeat(5-stars);
    const list = document.getElementById('reviews-list');
    if(!list) return;
    list.innerHTML = data.map(function(r){
        var name = escapeHtml(r.reviewer_name || 'Anonymous');
        var starsHtml = '⭐'.repeat(r.rating);
        var txt = r.review_text ? '<p style="font-size:13px;color:#334155;margin:0;font-family:sans-serif;">'+escapeHtml(r.review_text)+'</p>' : '';
        return '<div style="background:#f8fafc;border-radius:10px;padding:15px;border:1px solid #e2e8f0;margin-bottom:8px;"><div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:6px;"><strong style="font-size:14px;color:#0f172a;">'+name+'</strong><span>'+starsHtml+'</span></div>'+txt+'</div>';
    }).join('');
}
async function showPaymentView() {
    const {data:{session}} = await supabaseClient.auth.getSession();

    const {data: pending} = await supabaseClient

async function navigateChapter(dbId) {
    if(dbId < 1) return;
    showLoader();
    const {data: chapter} = await supabaseClient
        .from('chapters').select('is_premium').eq('id', dbId).single();
    if(!chapter){ hideLoader(); return; }

    if(chapter.is_premium){
        const {data:{session}} = await supabaseClient.auth.getSession();
        if(!session){ hideLoader(); return; }
        const {data: profile} = await supabaseClient
            .from('profiles').select('is_premium').eq('id', session.user.id).single();
        if(profile && profile.is_premium){
            const {data: cd, error} = await supabaseClient
                .from('chapters').select('content').eq('id', dbId).single();
            if(error || !cd){ hideLoader(); return; }
            currentChapterId = dbId; markChapterRead(dbId);
            const iframe = document.getElementById('chapter-fullscreen-iframe');
            iframe.onload = function(){ hideLoader(); };
            iframe.srcdoc = cd.content;
        } else {
            hideLoader();
            closeFullscreenChapter();
            showPaymentView();
        }
    } else {
        hideLoader();
        openFreeChapter(dbId);
    }
}
